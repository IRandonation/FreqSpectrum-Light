#include <FastLED.h>

#define NUM_LEDS 128 // 8*16 LED matrix
#define DATA_PIN 10   // 数据引脚连接到ESP32的第5号引脚
#define WIDTH 16     // 矩阵宽度
#define HEIGHT 8     // 矩阵高度
#define BRIGHTNESS 50 // LED亮度

CRGB leds[NUM_LEDS];

void setup() {
  FastLED.addLeds<WS2812, DATA_PIN, GRB>(leds, NUM_LEDS);
  FastLED.setBrightness(BRIGHTNESS);
}

void loop() {
  for (int i = 0; i < NUM_LEDS; i++) {
    // 依次点亮每个LED
    leds[i] = CRGB::White; // 设置当前LED为白色
    FastLED.show();        // 更新LEDs
    delay(100);            // 保持点亮状态一段时间
    leds[i] = CRGB::Black; // 关闭当前LED
  }
}
